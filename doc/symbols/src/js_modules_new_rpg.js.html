311     /** Battle step will cause a turn in a battle to take place
312      * @event
313      * @param {rpgCtx} ctx
314      * @param {rpgBattle} ctx.battleid The id of the specific battle this should step.
315      * @param {rpgClass} ctx.rpg The current rpg we are in.
316      */
317     battleStep: function (ctx)
318     {
319         var x;
320         var rpg = ctx.rpg;
321 
322         var battle = ctx.battle;
323 
324         // team_players is an array of all the players that are playing
325         // we make it from battle.players
326         var team_players = [];
327         for (var x in battle.players)
328         {
329             team_players.push(rpg.players[battle.players[x]]);
330             // Battle players contains NAMES of players, not the player objects!
331         }
332 
333         var team_mobs = []; // Do not save!
334         for ( x in battle.mobs)
335         {
336             team_mobs.push(battle.mobs[x]);
337         }
338 
339         var entities = []; // Do not save!
340 
341 
342         for (x in team_players)
343         {
344             entities.push({type: "player", e: team_players[x]});
345         }
346 
347         for (x in team_mobs)
348         {
349             entities.push({type: "mob", e: team_mobs[x]});
350         }
351 
352         entities.sort(
353             function _sorting_function_ (a, b)
354             {
355                 return a.speed - b.speed;
356             }
357         );
358 
359         var pids = [];
360 
361         for ( x in battle.players) pids.push(sys.id(battle.players[x]));
362 
363         battleLoop: for ( x in entities)
364         {
365             var attacker = ctx.attacker = entities[x].e;
366             //ctx.attackerIsPlayer = entities[x].type == "player";
367             var move = ctx.move = this.pickMove(entities[x]);
368 
369             if (ctx.move.cost) for (var x2 in ctx.move.cost)
370             {
371                 ctx.attacker[(x2 === "mp"? "mana" : x)] -= ctx.move.cost[x2];
372                 if (ctx.attacker[(x2 === "mp"? "mana" : x)] < 0)
373                 {
374                     this.com.message(pids, ctx.attacker.name + " tried to use "  + ctx.move.name + " but didn't have enough " + this.longStatName[x2 === "mp" ? "mana" : x2],
375                                     this.theme.GAME);
376                     continue battleLoop;
377                 }
378             }
379 
380             for (var x2 in ctx.move.components)
381             {
382                 var cmp = ctx.move.components[x2];
383                 var targets = ctx.targets = [];
384                 var count = cmp.count;
385                 switch (typeof cmp.target)
386                 {
387                 case "object":
388                     // Array
389                     if (cmp.target.indexOf("ally") != -1) ctx.targets = ctx.targets.concat(team_players);
390                     if (cmp.target.indexOf("opp") != -1) ctx.targets = ctx.targets.concat(team_mobs);
391                     if (cmp.target.indexOf("self") != -1 && ctx.indexOf(ctx.attacker) == -1) ctx.targets.push(ctx.attacker);
392                     break;
393                 case "string":
394                     switch (ctx.move.type)
395                     {
396                         case "ally": ctx.targets = team_players;
397                         case "self": ctx.targets = [attacker];
398                     }
399                 }
400 
401                 this.util.shuffle(targets);
402 
403                 if (count > 0) for (var x3 in targets)
404                 {
405                     if (count-- === 0) break;
406 
407 
408                     this.moves[cmp.move]({attacker: entities[x], targets:targets, component:cmp});
409 
410 
411 
412                 }
413             }
414         }
415 
416     }

462     entityTick: function (e)
463     {
464 
465         e.sta += e.maxsta/300;
466         e.msta += e.maxmsta/60;
467 
468         // Stamina
469         if (e.sp < 0)
470             // No stamina
471         {
472             e.hp += e.sp;
473             // Reduce the HP by the amount of missing stamina.
474 
475             e.sta = 0;
476             // Reset stamina
477         }
478         if (e.sp > e.maxsp / 3)
479             // Has a fairly large amount of stamina
480         {
481             hp += maxhp/600;
482             // Recover some HP
483         }
484         if (e.sp > e.maxsp)
485             // Too much stamina!
486         {
487             e.sp = e.maxsp;
488             // Remove overflow
489             
490             hp += maxhp/300;
491             // recover extra hp!
492         }
493 
494         // Mental Stamina
495         if (e.msta < 0)
496             // No mental stamina
497         {
498             e.mp -= e.msta;
499             e.hp -= e.msta;
500             // Lose hp and mana
501 
502             e.msta = 0;
503             // Reset
504         }
505         if (e.msta > e.maxmsta / 3)
506             // Recover mana
507         {
508             e.mana += e.maxmana/100;
509             // increase
510         }
511 
512         // Mana
513         if (e.mana < 0)
514             // Not enough mana
515         {
516             e.sta += e.mana;
517             e.msta += e.mana;
518             // Decrease staminas
519 
520             e.mana = 0;
521             // Reset mana;
522         }
523         if (e.mana > e.maxmana)
524             // Too much mana
525         {
526             if (e.mana > e.maxmana *1.1) e.hp += e.maxmana*1.1 - e.mana;
527             // Decrease HP by overflow, uncontrolable mana damages body!
528             
529             e.mana = e.maxmana;
530             // Reset
531         }
532 
533         // Health
534         if (e.hp <= 0 && !e.invincible && (!e.undead && e.hp > -e.maxhp*5))
535             // check for death
536         {
537             this.entityDie(e);
538         }
539         else if (e.hp > e.maxhp)
540             // overflow
541         {
542             e.hp = e.maxhp;
543             // reset
544         }
545         
546 
547         
548     }

1622     playerStep: function (player, ctx)
1623     {
1624         if (player.battling) return;
1625         // Regular player events don't occur while the player is in a battle!
1626 
1627         for (var x in player.activeActions)
1628         {
1629             if ("timer" in player.activeActions[x] && player.activeActions[x].timer-- <= 0)
1630             {
1631                 if (player.activeActions[x].done) this.actions[player.activeActions[x].done].call(this, player.activeActions[x], { rpg: ctx.rpg, player:player, index: x });
1632                 delete player.activeActions[x];
1633             }
1634             else if (player.activeActions[x].tick)
1635             {
1636                  this.actions[player.activeActions[x].tick].call(this, player.activeActions[x], { rpg: ctx.rpg, player:player, index: x });
1637             }
1638             break;
1639         }
1640 
1641         this.playerUpdateStats(player);
1642     }

1646     {
1647         e.maxmp =  (e.mag*0.13 + (Math.log(e.mag+Math.E)*10 | 0));
1648         e.maxsp = (e.str*0.02 + e.res*0.02 + (Math.log(e.res/100+Math.E)*150 | 0));
1649         e.maxmsp = (e.res*0.01 + e.mag*0.01 + e.psy*0.12 + e.spr*0.01 + (Math.log(e.psy/1000+Math.E)*50 | 0));
1650         e.maxhp = (e.str*0.01 + e.res*0.03 + (Math.log(e.res/100+Math.E)*100 | 0));
1651         e.power = Math.floor(800*Math.log(3/2*e.str+e.psy/3+Math.E)+e.psy/25000+300);
1652         e.offense = Math.floor(e.power / 10000 * (100 + this.equipAtk(e.lhand) + this.equipAtk(e.rhand)));
1653         e.defense = Math.floor(e.power / 10000 * (100 + this.equipDef(e.lhand)/2 + this.equipDef(e.rhand)/2 + this.equipDef(e.body) + this.equipDef(e.feet) + this.equipDef(e.head) + this.equipDef(e.back)));
